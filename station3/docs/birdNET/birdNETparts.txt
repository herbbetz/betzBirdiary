----audiorec.sh:
#!/bin/bash
# arecord wavs of 3 secs with a timestamp of secs:
arecord -f S16_LE -c1 -r48000 -t wav --max-file-time 3 -D default --use-strftime /home/pi/station3/ramdisk/audio-%s.wav
----audioclean.sh
#!/bin/bash
# each minute clean up wavs older than 5 mins (-> keep around 100 wavs of 3 secs):
find /home/pi/station3/ramdisk -maxdepth 1 -type f -name 'audio-*.wav' -mmin +5 -delete

#!/home/pi/audiovenv/bin/python3
import os
import re
import sys

import inotify.adapters
from inotify.constants import IN_CLOSE_WRITE

import msgBird as ms
from configBird3 import birdpath

WAV_RE = re.compile(r"^audio-\d+\.wav$")


def classify_wav(wav_path):
    # TODO: load WAV, run BirdNET/TFLite model, write result JSON/log
    ms.log(f"audio classify {wav_path}")


def main():
    ms.init()

    ramdisk = birdpath["ramdisk"]
    watcher = inotify.adapters.Inotify()
    watcher.add_watch(ramdisk, mask=IN_CLOSE_WRITE)

    ms.log(f"{sys.argv[0]} watching {ramdisk} for audio wav chunks")

    try:
        for event in watcher.event_gen(yield_nones=False):
            (_, type_names, path, filename) = event

            if not WAV_RE.match(filename):
                continue

            wav_path = os.path.join(path, filename)

            if not os.path.exists(wav_path):
                continue

            if os.path.getsize(wav_path) == 0:
                ms.log(f"skip empty wav {wav_path}")
                continue

            classify_wav(wav_path)

    except Exception as e:
        ms.log(f"Exception in {sys.argv[0]}: {e}")
        raise


if __name__ == "__main__":
    main()
