---- bash example:
# Function to create wlan.yml from current settings
create_wlan_yml() {
    local con_name="$1"
    local old_con="$2"

    echo "Creating '$WLAN_YML' from current nmcli settings..."
    {
        echo "con_name: $con_name"
        echo "old_con: $old_con"
        echo "ssid: $(nmcli -g 802-11-wireless.ssid connection show "$con_name" || echo "MyNetwork")"
        echo "psk: $(nmcli -g 802-11-wireless-security.psk connection show "$con_name" || echo "changeme123")"
        local ip=$(nmcli -g ipv4.addresses connection show "$con_name" | cut -d/ -f1)
        echo "ipaddr: ${ip:-192.168.178.210}"
        echo "gateway: $(nmcli -g ipv4.gateway connection show "$con_name" || echo "192.168.178.1")"
    } > "$WLAN_YML"

    echo "'$WLAN_YML' created."
    exit 0
}
